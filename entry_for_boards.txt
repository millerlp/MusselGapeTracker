Copy the section between the hash marks below into the file called
boards.txt found in your Arduino installation

################################################################
# For Arduino IDE v 1.6.5 or later
# Tested successfully using Arduino as ISP programmer to burn
# bootloader on MusselGapeTracker.
# Cut and paste this entry into your existing boards.txt file 
# found inside the /hardware/arduino/avr/ directory. 
# Select the MusselGapeTracker (8MHz internal clock) entry in the
# Arduino IDE menu under Tools>Board>

MusselGapeTracker328.name=MusselGapeTracker (8 MHz internal clock)
MusselGapeTracker328.upload.tool=arduino:avrdude
MusselGapeTracker328.upload.protocol=arduino
MusselGapeTracker328.upload.maximum_size=30720
MusselGapeTracker328.upload.speed=57600

MusselGapeTracker328.bootloader.low_fuses=0xE2
MusselGapeTracker328.bootloader.high_fuses=0xDA
MusselGapeTracker328.bootloader.extended_fuses=0x05 

MusselGapeTracker328.bootloader.file=atmega/ATmegaBOOT_168_atmega328_pro_8MHz.hex
MusselGapeTracker328.bootloader.unlock_bits=0x3F
MusselGapeTracker328.bootloader.lock_bits=0x0F
MusselGapeTracker328.bootloader.tool=arduino:avrdude
MusselGapeTracker328.build.mcu=atmega328p
MusselGapeTracker328.build.f_cpu=8000000L
MusselGapeTracker328.build.core=arduino
MusselGapeTracker328.build.board=AVR_MUSSELGAPETRACKER
MusselGapeTracker328.build.variant=standard 

##############################################################
